  { vinoId: 1, sirId: 1 },
  { vinoId: 1, sirId: 3 },
  { vinoId: 1, sirId: 5 },

  { vinoId: 2, sirId: 20 },
  { vinoId: 2, sirId: 6 },
];
🔁 IZ TOGA GENERIRAŠ OBA PRIKAZA
🍷 vina → sirevi
function getSireviZaVino(vinoId) {
  return pairings
    .filter(p => p.vinoId === vinoId)
    .map(p => p.sirId);
}
🧀 sirevi → vina
function getVinaZaSir(sirId) {
  return pairings
    .filter(p => p.sirId === sirId)
    .map(p => p.vinoId);
}
